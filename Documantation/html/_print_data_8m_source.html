208 - (void) removeImagesWithUrls:(NSArray *)urls
209 {
211  int (^GetIndexWithSavedImages)(NSString *, NSMutableArray *) = ^(NSString *removeUrl, NSMutableArray *savedImages) {
212  NSPredicate *prediate = [NSPredicate predicateWithFormat:@"urlLibrary == %@", removeUrl];
213  PrintImage *printImage = [[savedImages filteredArrayUsingPredicate:prediate] firstObject];
214  if (!printImage) {
215  return -1;
216  } else {
217  NSInteger index = [savedImages indexOfObject:printImage];
218  return (int)index;
219  }
220  };
221 
222  // Filtred
223  NSMutableArray *mutableImages = [_strategy.imagesPreview mutableCopy];
224  NSMutableIndexSet *mutableSet = [NSMutableIndexSet indexSet];
225  for (NSString *removeUrl in urls) {
226  NSInteger index = GetIndexWithSavedImages(removeUrl, mutableImages);
227  if (index >= 0) {
228  [mutableSet addIndex:index];
229  }
230  }
231 
232  // Remove
233  [mutableImages removeObjectsAtIndexes:mutableSet];
234  _strategy.images = [mutableImages copy];
235 }

251 - (void) replacePreviewPrintImage:(PrintImage *)printImage
252 {
253  // Если это скленная картинка, то не добавляем
254  if (printImage.isMergedImage) {
255  return;
256  }
257 
258  NSMutableArray *mutableAray = [_strategy.images mutableCopy];
259  BOOL imageFinded = NO;
260 
261  for (int imageIndex = 0; imageIndex < mutableAray.count; imageIndex++) {
262  PrintImage *image = [mutableAray objectAtIndex:imageIndex];
263  if ([image.urlLibrary isEqualToString:printImage.urlLibrary]) {
264  imageFinded = YES;
265  [mutableAray replaceObjectAtIndex:imageIndex withObject:printImage];
266  }
267  }
268 
269  if (!imageFinded) {
270  NSLog(@"Image Do not finded");
271  return;
272  }
273 
274 
275  _strategy.images = [mutableAray copy];
276

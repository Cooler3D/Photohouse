38 +(UIImage *)applyFilter:(UIImage *)image withFilterName:(NSString *)filterName {
39  if (!filterName) {
40  return image;
41  }
42 
43  NSDictionary *filterProperty = [[self defaultFilterInfo] objectForKey:filterName];
44  NSString *fName = [filterProperty objectForKey:@"name"];
45 
46 
47 
48  if([fName isEqualToString:@"CLDefaultEmptyFilter"]){
49  return image;
50  }
51 
52  CIImage *ciImage = [[CIImage alloc] initWithImage:image];
53  CIFilter *filter = [CIFilter filterWithName:fName keysAndValues:kCIInputImageKey, ciImage, nil];
54 
55  //NSLog(@"%@", [filter attributes]);
56 
57  [filter setDefaults];
58 
59  if([fName isEqualToString:@"CIVignetteEffect"]){
60  // parameters for CIVignetteEffect
61  CGFloat R = MIN(image.size.width, image.size.height)/2;
62  CIVector *vct = [[CIVector alloc] initWithX:image.size.width/2 Y:image.size.height/2];
63  [filter setValue:vct forKey:@"inputCenter"];
64  [filter setValue:[NSNumber numberWithFloat:0.9] forKey:@"inputIntensity"];
65  [filter setValue:[NSNumber numberWithFloat:R] forKey:@"inputRadius"];
66  }
67 
68  CIContext *context = [CIContext contextWithOptions:nil];
69  CIImage *outputImage = [filter outputImage];
70  CGImageRef cgImage = [context createCGImage:outputImage fromRect:[outputImage extent]];
71 
72  UIImage *result = [UIImage imageWithCGImage:cgImage];
73 
74  CGImageRelease(cgImage);
75 
76  return result;
77 }

57 - (UIImage*)filteredImage:(UIImage*)image withFilterName:(NSString*)filterName
58 {
59  if([filterName isEqualToString:@"CLDefaultEmptyFilter"] || !filterName){
60  return image;
61  }
62 
63  CIImage *ciImage = [[CIImage alloc] initWithImage:image];
64  NSString *fName = [self filterName:filterName];
65  CIFilter *filter = [CIFilter filterWithName:fName keysAndValues:kCIInputImageKey, ciImage, nil];
66 
67  //NSLog(@"%@", [filter attributes]);
68 
69  [filter setDefaults];
70 
71  if([filterName isEqualToString:@"CIVignetteEffect"]){
72  // parameters for CIVignetteEffect
73  CGFloat R = MIN(image.size.width, image.size.height)/2;
74  CIVector *vct = [[CIVector alloc] initWithX:image.size.width/2 Y:image.size.height/2];
75  [filter setValue:vct forKey:@"inputCenter"];
76  [filter setValue:[NSNumber numberWithFloat:0.9] forKey:@"inputIntensity"];
77  [filter setValue:[NSNumber numberWithFloat:R] forKey:@"inputRadius"];
78  }
79 
80  CIContext *context = [CIContext contextWithOptions:nil];
81  CIImage *outputImage = [filter outputImage];
82  CGImageRef cgImage = [context createCGImage:outputImage fromRect:[outputImage extent]];
83 
84  UIImage *result = [UIImage imageWithCGImage:cgImage];
85 
86  CGImageRelease(cgImage);
87 
88  return result;
89 }

93 - (UIImage*)contrastImage:(UIImage*)image andSetting:(EditImageSetting *)setting
94 {
95  // Is Default
96  if (setting.contrastValue == DEFAULT_CONSTRAST &&
97  setting.brightnessValue == DEFAULT_BRIGHTNESS &&
98  setting.saturationValue == DEFAULT_SATURATION)
99  {
100  return image;
101  }
102 
103 
104  CIImage *ciImage = [[CIImage alloc] initWithImage:image];
105  CIFilter *filter = [CIFilter filterWithName:@"CIColorControls" keysAndValues:kCIInputImageKey, ciImage, nil];
106 
107  [filter setDefaults];
108  [filter setValue:[NSNumber numberWithFloat:setting.saturationValue] forKey:@"inputSaturation"];
109 
110  filter = [CIFilter filterWithName:@"CIExposureAdjust" keysAndValues:kCIInputImageKey, [filter outputImage], nil];
111  [filter setDefaults];
112  CGFloat brightness = 2*setting.brightnessValue;
113  [filter setValue:[NSNumber numberWithFloat:brightness] forKey:@"inputEV"];
114 
115  filter = [CIFilter filterWithName:@"CIGammaAdjust" keysAndValues:kCIInputImageKey, [filter outputImage], nil];
116  [filter setDefaults];
117  CGFloat contrast = setting.contrastValue*setting.contrastValue;
118  [filter setValue:[NSNumber numberWithFloat:contrast] forKey:@"inputPower"];
119 
120  CIContext *context = [CIContext contextWithOptions:nil];
121  CIImage *outputImage = [filter outputImage];
122  CGImageRef cgImage = [context createCGImage:outputImage fromRect:[outputImage extent]];
123 
124  UIImage *result = [UIImage imageWithCGImage:cgImage];
125 
126  CGImageRelease(cgImage);
127 
128  return result;
129 }
